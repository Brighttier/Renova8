## Overview

### Purpose
This document outlines the comprehensive platform administration system for Renova8, enabling administrators to:
- Monitor and manage platform revenue streams
- Track and manage user accounts and subscriptions
- Oversee website hosting for customer-generated sites
- Analyze platform usage and performance metrics
- Monitor frontend and backend application health
- Handle user support requests efficiently

### Admin Roles & Permissions

| Role | Permissions |
|------|-------------|
| **Super Admin** | Full access to all systems, billing, user management, and configurations |
| **Finance Admin** | Revenue dashboards, billing management, refund processing |
| **Support Admin** | Support tickets, user assistance, limited user management |
| **Technical Admin** | Performance monitoring, hosting management, system health |
| **Analytics Admin** | Reports, analytics dashboards, data exports |

// Custom Traces for Key Operations

  const customTrace = trace(perf, 'website_generation');
  customTrace.start();

  try {
    const result = await generateWebsiteStructure(prompt);
    customTrace.putAttribute('success', 'true');
    customTrace.putMetric('code_length', result.length);
    return result;
  } catch (error) {
    customTrace.putAttribute('success', 'false');
    customTrace.putAttribute('error', error.message);
    throw error;
  } finally {
    customTrace.stop();
  }
};

// Track API Calls
export const trackAPICall = async (apiName: string, apiCall: () => Promise<any>) => {
  const customTrace = trace(perf, `api_${apiName}`);
  customTrace.start();

  try {
    const result = await apiCall();
    customTrace.putAttribute('status', 'success');
    return result;
  } catch (error) {
    customTrace.putAttribute('status', 'error');
    customTrace.putAttribute('error_type', error.name);
    throw error;
  } finally {
    customTrace.stop();
  }
};

  id: string;
  ticketNumber: number; // Auto-increment: 1247, 1248, etc.

  // Ticket Info
  subject: string;

  priority: 'urgent' | 'high' | 'normal' | 'low';
  status: 'open' | 'in_progress' | 'awaiting_reply' | 'resolved' | 'closed';

  // User Info
  userId: string;
  userEmail: string;
  userName: string;

  // Assignment
  assigneeId?: string;
  assigneeName?: string;

  // Tags
  tags: string[];

  // Timestamps
  createdAt: Timestamp;
  updatedAt: Timestamp;
  firstResponseAt?: Timestamp;
  resolvedAt?: Timestamp;

  // SLA Tracking
  slaFirstResponse: Timestamp; // Deadline
  slaResolution: Timestamp; // Deadline
  slaBreached: boolean;

  // Satisfaction
  satisfactionRating?: 1 | 2 | 3 | 4 | 5;
  satisfactionFeedback?: string;
}

    name: string;
    url: string;
    type: string;
    size: number;
  }[];

// functions/src/alerts.ts
import * as functions from 'firebase-functions';
import * as admin from 'firebase-admin';

// Alert on high error rate
export const errorRateAlert = functions.pubsub
  .schedule('every 5 minutes')
  .onRun(async (context) => {
    const errors = await getRecentErrors(5); // Last 5 minutes

    if (errors.length > 20) {
      await sendAlert({
        type: 'high_error_rate',
        severity: 'critical',
        message: `High error rate detected: ${errors.length} errors in last 5 minutes`,
        channels: ['slack', 'email'],
      });
    }
  });

// Alert on revenue anomaly
export const revenueAlert = functions.pubsub
  .schedule('every day 09:00')
  .timeZone('America/New_York')
  .onRun(async (context) => {
    const yesterday = await getDailyRevenue(1);
    const average = await getAverageRevenue(30);

    if (yesterday < average * 0.7) {
      await sendAlert({
        type: 'revenue_drop',
        severity: 'high',
        message: `Revenue alert: Yesterday's revenue ($${yesterday}) is 30% below average ($${average})`,
        channels: ['email'],
      });
    }
  });

// Alert on urgent support ticket
export const urgentTicketAlert = functions.firestore
  .document('support_tickets/{ticketId}')
  .onCreate(async (snap, context) => {
    const ticket = snap.data();

    if (ticket.priority === 'urgent') {
      await sendAlert({
        type: 'urgent_ticket',
        severity: 'high',
        message: `Urgent support ticket #${ticket.ticketNumber}: ${ticket.subject}`,
        channels: ['slack'],
        data: { ticketId: context.params.ticketId },
      });
    }
  });

## Appendix

### A. Firestore Security Rules

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admin-only collections
    match /admin_users/{userId} {
      allow read, write: if isAdmin();
    }

    match /audit_logs/{logId} {
      allow read: if isAdmin();
      allow write: if false; // Only via Cloud Functions
    }

    match /support_tickets/{ticketId} {
      allow read: if isAdmin() || request.auth.uid == resource.data.userId;
      allow create: if request.auth != null;
      allow update: if isAdmin();
    }

    // Helper functions
    function isAdmin() {
      return request.auth != null &&
             get(/databases/$(database)/documents/admin_users/$(request.auth.uid)).data.role in ['super_admin', 'admin'];
    }
  }
}
```
